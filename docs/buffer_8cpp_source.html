45 void BufMgr::allocBuf(FrameId &frame)
46 {
47  bool openFrameFound = false;
48  int counter = -1;
49  while (counter < (int)numBufs && openFrameFound == false)
50  {
51  counter++;
52  advanceClock();
53  if (bufDescTable[clockHand].valid == true)
54  {
55  if (bufDescTable[clockHand].refbit == true)
56  {
57  bufDescTable[clockHand].refbit = false;
58  continue; // advance clock and try again
59  }
60  else if (bufDescTable[clockHand].pinCnt > 0)
61  {
62  continue; // advance clock and try again
63  }
64  else if (bufDescTable[clockHand].refbit == false && bufDescTable[clockHand].pinCnt == 0)
65  {
66  // Use the frame
67  frame = clockHand;
68 
69  if (bufDescTable[clockHand].dirty)
70  {
71  // Flush page to disk
72  bufDescTable[clockHand].file.writePage(bufPool[frame]);
73  }
74  hashTable.remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
75  }
76  openFrameFound = true;
77  bufDescTable[clockHand].clear();
78  }
79  else
80  {
81  // Use the frame
82  frame = clockHand;
83  openFrameFound = true;
84  bufDescTable[clockHand].clear();
85  }
86  }
87  if (openFrameFound == false)
88  {
89  throw BufferExceededException();
90  }

133 {
134  try
135  {
136  // Check if page is in hashTable
137  FrameId frameNum; // to be replaced by the hashTable.lookup

139

192  // for (BufDesc frame : bufDescTable)
193  for (u_int32_t i = 0; i < numBufs; i++)
194  {

202  // Throws PagePinnedException if some page of the file is pinned.
203  if (bufDescTable[i].pinCnt > 0)
204  {
205  throw PagePinnedException(bufDescTable[i].file.filename(), bufDescTable[i].pageNo, bufDescTable[i].frameNo);
206  }
207  if (bufDescTable[i].dirty)
208  {
209  // if the page is dirty, call file.writePage() to flush the page to disk
210  // and then set the dirty bit for the page to false
211

213  // bufDescTable[i].file.writePage(bufDescTable[i].pageNo, bufDescTable[i].pageNo);
214  bufDescTable[i].file.writePage(bufPool[i]);
215  bufDescTable[i].dirty = false;
216  }
217  // Throws BadBufferException if an invalid page belonging to the file is encountered
218  if (Page::INVALID_NUMBER == bufDescTable[i].pageNo)
219  {
220  throw BadBufferException(bufDescTable[i].frameNo, bufDescTable[i].dirty, bufDescTable[i].valid, bufDescTable[i].refbit);
221  }
222  // remove the page from the hashtable (whether the page is clean or dirty)
223  hashTable.remove(bufDescTable[i].file, bufDescTable[i].pageNo);
224
